# Editable install of every component in this monorepo.
#
# Real dependency lists live in each component's pyproject.toml.
# This file just chains `pip install -e .` across all of them so a
# new developer can spin up a single dev venv with one command:
#
#   pip install -r requirements.txt
#
# Production containers do NOT use this file — each service's
# Dockerfile installs only its own component via `pip install .`.

# Shared library (other components depend on it transitively).
# `[migrations]` extras pull SQLAlchemy + alembic + psycopg2 for dev — runtime
# services do not need them, only the migrations one-shot container does.
-e ./common[migrations]

# API gateway (HTTP entry point).
-e ./gateway

# Workers (RabbitMQ consumers).
-e ./services/auth
-e ./services/attachments
-e ./services/comments
-e ./services/tags
-e ./services/tasks
-e ./services/users
